
docker run --name fc-mysql -p 3313:3306 -e MYSQL_ROOT_PASSWORD=mysql1234 -e MYSQL_DATABASE=fertilizer-calculator -e MYSQL_USER=mysqlfc -e MYSQL_PASSWORD=mysql1234 -d mysql


Пояснение:
--name fc-mysql: Имя контейнера (fc-mysql).
-p 3313:3306: Проброс порта, 3306 — это стандартный порт для MySQL.
-e MYSQL_ROOT_PASSWORD=mysql1234: Устанавливает пароль для пользователя root.
-e MYSQL_DATABASE=fertilizer-calculator: Создаёт базу данных с именем fertilizer-calculator.
-e MYSQL_USER=mysqlfc: Создаёт пользователя с именем mysqlfc.
-e MYSQL_PASSWORD=mysql1234: Устанавливает пароль для указанного пользователя.
После запуска контейнера MySQL вы можете подключиться к базе данных через клиент MySQL или другие инструменты, используя указанные параметры.

DATABASE_PORT=3313;
DATABASE_NAME=fertilizer-calculator;
DATABASE_USERNAME=mysqlfc;
DATABASE_PASSWORD=mysql1234;
